01
SETUP
Setup & Configuration
// initialize repos and set your identity
git configEssential

Set your identity and preferences globally or per-repo

$git config --global user.name "Name"
$git config --global user.email "you@mail.com"
$git config --list

use --local to override per project

git initEssential

Create a brand new local repository

$git init
$git init my-project

creates a hidden .git directory

git cloneEssential

Copy a remote repository to your machine

$git clone https://github.com/user/repo.git
$git clone <url> folder-name

downloads full history and all branches

git helpUtility

Open documentation for any command

$git help commit
$git commit --help

opens man page in browser or terminal

02
STAGING
Staging & Committing
// stage changes and create commits
git addEssential

Stage files for your next commit

$git add .
$git add filename.js
$git add -p

-p lets you stage changes interactively

git commitEssential

Save staged changes as a snapshot

$git commit -m "feat: add login"
$git commit -am "fix: typo"
$git commit --amend

--amend rewrites the last commit

git statusEssential

See what's changed in your working tree

$git status
$git status -s

-s shows compact short output

git diffUtility

Show line-level changes between states

$git diff
$git diff --staged
$git diff main..feature

--staged compares staged vs last commit

git mvUtility

Rename or move a tracked file

$git mv old.js new.js

auto-stages the rename operation

git rmCaution

Stop tracking a file and optionally delete it

$git rm filename.js
$git rm --cached filename.js

--cached keeps the local file

03
BRANCHING
Branching & Merging
// parallel development workflows
git branchEssential

Create, list, rename, and delete branches

$git branch
$git branch feature/login
$git branch -d branch-name

-D forces delete an unmerged branch

git checkoutEssential

Switch branches or restore working files

$git checkout main
$git checkout -b feature/new

-b creates and switches in one command

git switchModern

Dedicated command to switch branches (v2.23+)

$git switch main
$git switch -c feature/new

clearer intent than checkout

git mergeEssential

Join two or more branches together

$git merge feature/login
$git merge --no-ff feature/login

--no-ff always creates a merge commit

git rebaseAdvanced

Replay commits on top of another branch

$git rebase main
$git rebase -i HEAD~3

-i opens interactive squash/reorder mode

git cherry-pickAdvanced

Apply a specific commit from another branch

$git cherry-pick abc1234
$git cherry-pick a1..b2

great for backporting hotfixes

git tagRelease

Mark release points in history

$git tag v1.0.0
$git tag -a v1.0.0 -m "release"
$git push origin v1.0.0

use semver: major.minor.patch

04
REMOTE
Remote Workflows
// collaborate with remote repositories
git remoteEssential

Manage remote repository connections

$git remote -v
$git remote add origin <url>
$git remote remove origin

-v shows fetch and push URLs

git pushEssential

Upload your commits to a remote repository

$git push origin main
$git push -u origin feature
$git push --force-with-lease

--force-with-lease is safer than --force

git pullEssential

Fetch and integrate remote changes

$git pull origin main
$git pull --rebase

--rebase avoids merge commit clutter

git fetchSafe

Download remote changes without merging

$git fetch origin
$git fetch --all --prune

--prune removes deleted remote branches

git submoduleAdvanced

Embed another repository inside yours

$git submodule add <url> path/
$git submodule update --init

great for managing shared libraries

05
UNDOING
Undoing Changes
// recover, revert, and reset your work
git resetCaution

Move HEAD and optionally undo commits

$git reset HEAD~1
$git reset --soft HEAD~1
$git reset --hard HEAD~1

--hard deletes changes permanently

git revertSafe

Undo a commit by creating an inverse commit

$git revert abc1234
$git revert HEAD

preferred when commits are already pushed

git restoreModern

Discard changes in working tree or index

$git restore file.js
$git restore --staged file.js

cleaner replacement for checkout in v2.23+

git stashUtility

Shelve changes to revisit later

$git stash
$git stash pop
$git stash list

stash push -m "name" to label stashes

git cleanCaution

Delete all untracked files and directories

$git clean -fd
$git clean -n

-n does a dry run first — use it!

06
INSPECTION
Inspection & History
// explore commits and blame the right person
git logEssential

Explore the full commit history

$git log --oneline
$git log --graph --all --oneline
$git log --author="Name" --since="2w"

combine flags for powerful log filtering

git showUtility

Inspect a commit, tag, or object

$git show abc1234
$git show HEAD:file.js

shows patch + commit info

git blameUtility

See who last modified each line

$git blame file.js
$git blame -L 10,20 file.js

-L limits blame to a line range

git bisectAdvanced

Binary-search history to find a bad commit

$git bisect start
$git bisect good v1.0.0
$git bisect bad HEAD

git bisect reset to end the session

git shortlogUtility

Summarize commits per author

$git shortlog -sn

-s=count only, -n=sort by count

07
ADVANCED
Advanced Operations
// for when you mean business
git reflogRecovery

See every HEAD movement — nothing is truly lost

$git reflog
$git checkout HEAD@{2}

your ultimate undo — expires after 90 days

git worktreeAdvanced

Check out multiple branches simultaneously

$git worktree add ../hotfix hotfix/v2
$git worktree list

one repo, many working trees

git archiveUtility

Export a snapshot as zip or tarball

$git archive --format=zip HEAD > out.zip

no .git folder — clean deploy artifact

git gcUtility

Clean up and compress repository internals

$git gc --aggressive

run on large repos to reduce disk size

git hooksAdvanced

Run scripts automatically on Git events

$ls .git/hooks/
$chmod +x .git/hooks/pre-commit

use husky or lefthook for team hooks

08
PATCHING
Patching
// create and apply patch files
git applyUtility

Apply a patch file without committing

$git apply changes.patch
$git apply --check changes.patch

--check tests if patch will apply cleanly

git format-patchAdvanced

Generate patch files from commit range

$git format-patch main
$git format-patch -1 HEAD

-1 creates patch for just the last commit

git amAdvanced

Apply a patch as a commit with author info

$git am changes.patch

preserves commit authorship — better than apply
